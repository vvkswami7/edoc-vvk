#include <stdio.h>
#include <string.h>

int main(){
    char names[5][20] = {"Amit","Ravi","Sita","Gita","Kiran"};
    char key[20];
    printf("Enter name to search: ");
    scanf("%s", key);
    for(int i=0;i<5;i++)
        if(strcmp(names[i],key)==0){
            printf("Found at position %d\n",i+1);
            return 0;
        }
    printf("Not found\n");
    return 0;
}

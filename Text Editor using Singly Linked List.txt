#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Node {
    char line[100];
    struct Node* next;
};

struct Node* head = NULL;

void insertEnd(char text[]) {
    struct Node* newNode = malloc(sizeof(struct Node));
    strcpy(newNode->line, text);
    newNode->next = NULL;
    if(head == NULL) head = newNode;
    else {
        struct Node* temp = head;
        while(temp->next) temp = temp->next;
        temp->next = newNode;
    }
}

void deleteFront() {
    if(head == NULL) printf("List Empty\n");
    else {
        struct Node* temp = head;
        head = head->next;
        free(temp);
    }
}

void display() {
    struct Node* temp = head;
    while(temp) {
        printf("%s\n", temp->line);
        temp = temp->next;
    }
}

int main() {
    int ch;
    char text[100];
    do {
        printf("\n1.Insert Line 2.Delete Front 3.Display 4.Exit: ");
        scanf("%d", &ch);
        getchar();
        if(ch == 1) { printf("Enter text: "); gets(text); insertEnd(text); }
        else if(ch == 2) deleteFront();
        else if(ch == 3) display();
    } while(ch != 4);
    return 0;
}

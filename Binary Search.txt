#include <stdio.h>
#include <string.h>

int main(){
    char names[5][20] = {"Amit","Gita","Kiran","Ravi","Sita"};
    char key[20];
    int low=0, high=4, mid;
    printf("Enter name: ");
    scanf("%s", key);
    while(low<=high){
        mid=(low+high)/2;
        int res=strcmp(key,names[mid]);
        if(res==0){ printf("Found at position %d\n",mid+1); return 0; }
        else if(res<0) high=mid-1;
        else low=mid+1;
    }
    printf("Not Found\n");
    return 0;
}

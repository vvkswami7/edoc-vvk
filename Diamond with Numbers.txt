#include <stdio.h>

int main() {
    int rows;
    
    printf("Enter number of rows for upper half: ");
    scanf("%d", &rows);
    
    printf("\nDiamond Pattern:\n\n");
    
    // Upper half including middle
    for(int i = 1; i <= rows; i++) {
        // Print spaces
        for(int j = 1; j <= rows - i; j++) {
            printf(" ");
        }
        // Print numbers
        for(int j = 1; j <= i; j++) {
            printf("%d ", i);
        }
        printf("\n");
    }
    
    // Lower half
    for(int i = rows - 1; i >= 1; i--) {
        // Print spaces
        for(int j = 1; j <= rows - i; j++) {
            printf(" ");
        }
        // Print numbers
        for(int j = 1; j <= i; j++) {
            printf("%d ", i);
        }
        printf("\n");
    }
    
    return 0;
}

#include <stdio.h>
#define MAX 5

int stack[MAX], top = -1;

void push(int x) {
    if(top == MAX-1) printf("Stack Overflow\n");
    else stack[++top] = x;
}

void pop() {
    if(top == -1) printf("Stack Underflow\n");
    else printf("Popped: %d\n", stack[top--]);
}

void display() {
    if(top == -1) printf("Empty Stack\n");
    else for(int i = top; i >= 0; i--) printf("%d ", stack[i]);
}

int main() {
    int ch, x;
    do {
        printf("\n1.Push 2.Pop 3.Display 4.Exit: ");
        scanf("%d", &ch);
        if(ch == 1){ printf("Enter value: "); scanf("%d", &x); push(x);}
        else if(ch == 2) pop();
        else if(ch == 3) display();
    } while(ch != 4);
    return 0;
}

#include <stdio.h>

int main() {
    int rows;
    
    printf("Enter number of rows: ");
    scanf("%d", &rows);
    
    printf("\nAlphabet Pyramid Pattern:\n\n");
    
    for(int i = 1; i <= rows; i++) {
        // Print spaces
        for(int j = 1; j <= rows - i; j++) {
            printf(" ");
        }
        
        // Print ascending letters
        for(int j = 0; j < i; j++) {
            printf("%c ", 'A' + j);
        }
        
        // Print descending letters
        for(int j = i - 2; j >= 0; j--) {
            printf("%c ", 'A' + j);
        }
        
        printf("\n");
    }
    
    return 0;
}
